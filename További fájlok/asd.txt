using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Diagnostics;
using System.Drawing;
using System.IO;
using System.Linq;
using System.Net.NetworkInformation;
using System.Runtime.ConstrainedExecution;
using System.Runtime.InteropServices.Expando;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;
using static System.Windows.Forms.VisualStyles.VisualStyleElement;
namespace pizza
{
public partial class From : Form
    {
        string[] sorok;
        String[,] pizzak = new String[100, 100];
        CheckBox[,] checkBox = new CheckBox[100,100];
        public From()
        {         

            InitializeComponent();

        }

        private void Form1_Load(object sender, EventArgs e)
        {

        }

        private void btn_data_Click(object sender, EventArgs e)
        {
            

            OpenFileDialog openFileDialog = new OpenFileDialog();
            if (openFileDialog.ShowDialog() == DialogResult.OK)
            {
               sorok = File.ReadAllLines(openFileDialog.FileName);
            }
            int id = 0;

            foreach (string sor in sorok)
            {
                string[] parts = sor.Split(';');

                id++;
                string pizza = parts[0];
                int kicsi = int.Parse(parts[1]);
                int nagy = int.Parse(parts[2]);

                for (int i = id; i < id+1; i++)
                {
                        pizzak[i, 0] = pizza;
                        pizzak[i, 1] = kicsi.ToString();
                        pizzak[i, 2] = nagy.ToString();
                }

            }
            btn_data.Enabled = false;
            btn_data.Visible = false;
            BackgroundImage = null;
            BackColor = Color.Beige;
            textBox1.Visible = true;
            btn_delete.Visible = true;
            btn_sum.Visible = true;
            fizetendo.Visible = true;
            ar.Visible = true;
            btn_close.Visible = true;


            Debug.WriteLine(pizzak[1, 0]);
            textBox2.Visible = true;
            checkBox1.Visible = true;
            radioButton1.Visible = true;
            radioButton2.Visible = true;
            textBox2.Visible = true;
            textBox3.Visible = true;
            for (int i = 1; i < 2; i++)
            {
                checkBox1.Text = $"{pizzak[i, 0]}";
                radioButton1.Text = $"{pizzak[i, 1]} Ft";
                radioButton2.Text = $"{pizzak[i, 2]} Ft";
                
            }

            for (int i = 0; i < id; i++)
            {
                CheckBox checkBox3;
                checkBox3 = new System.Windows.Forms.CheckBox();
                checkBox3.Location = new System.Drawing.Point(200, 226);
                checkBox3.Name = "checkBox1";
                checkBox3.Size = new System.Drawing.Size(80, 17);
                checkBox3.Text = $"{pizzak[i, 0]}";
                checkBox3.Visible = true;
                checkBox3.Enabled = true;


            }

                    }

        private void openFileDialog1_FileOk(object sender, CancelEventArgs e)
        {

        }

        private void btn_close_Click(object sender, EventArgs e)
        {
            btn_data.Enabled = true;
            btn_data.Visible = true;
            BackgroundImage = global::pizza.Properties.Resources.pizzeria;
            textBox1.Visible = false;
            btn_delete.Visible = false;
            btn_sum.Visible = false;
            fizetendo.Visible = false;
            ar.Visible = false;
            btn_close.Visible = false;
            radioButton1.Visible = false;
            radioButton2.Visible = false;
            textBox2.Visible = false;
            textBox3.Visible = false;
            checkBox1.Visible = false;

        }

        private void btn_sum_Click(object sender, EventArgs e)
        {
            if (checkBox1.Checked == true && radioButton1.Checked == true)
            {
                int kicsi = int.Parse(pizzak[1, 1]);
                ar.Text = $"{kicsi * int.Parse(textBox2.Text)} Ft";
            }
            else if (checkBox1.Checked == true && radioButton2.Checked == true)
            {
                int nagy = int.Parse(pizzak[1, 2]);
                ar.Text = $"{nagy * int.Parse(textBox2.Text)} Ft";
            }
        }
